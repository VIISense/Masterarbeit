$c$: \textbf{set constraint} or a \textbf{cardinality constraint}
\end{frame}
\begin{frame}\frametitle{$\mathcal{ALCSCC}$: constraints}
\begin{minipage}[t]{0.5\textwidth}
\raggedright
\textbf{set constraint:}
\begin{itemize}
\item $r\subseteq s$
\item $C\cap r\subseteq D$
\item $succ(C\cap r)\subseteq succ(D)$
\end{itemize}
\end{minipage}%
\begin{minipage}[t]{0.5\textwidth}
\raggedright
\textbf{cardinality constraint}
\begin{itemize}
\item $3\,dvd\,|r|$
\item $|C\cap r|\leq |D|$
\item $|succ(C\cap r)|\leq |succ(D)|$
\end{itemize}
\end{minipage}
\end{frame}
